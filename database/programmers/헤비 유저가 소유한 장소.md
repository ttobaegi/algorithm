

```sql
/*
-- 헤비 유저가 소유한 장소
-- 헤비 유저 : 공간을 둘 이상 등록 -> COUNT (ID)
*/
-- 헤비 유저가 등록한 공간의 정보를 아이디 순으로 조회
-- ID | NAME | HOST_ID

WITH CTE1 AS (
    SELECT 
        HOST_ID, 
        COUNT(ID) AS `CNT`
    FROM PLACES
    GROUP BY 1 
)
SELECT *
FROM PLACES
-- 헤비유저 
WHERE HOST_ID IN (SELECT HOST_ID FROM CTE1 WHERE CNT >=2)
ORDER BY ID
```
